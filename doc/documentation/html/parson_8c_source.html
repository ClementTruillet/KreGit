214 static int verify_utf8_sequence(const unsigned char *string, int *len) {
215  unsigned int cp = 0;
216  *len = num_bytes_in_utf8_sequence(string[0]);
217 
218  if (*len == 1) {
219  cp = string[0];
220  } else if (*len == 2 && IS_CONT(string[1])) {
221  cp = string[0] & 0x1F;
222  cp = (cp << 6) | (string[1] & 0x3F);
223  } else if (*len == 3 && IS_CONT(string[1]) && IS_CONT(string[2])) {
224  cp = ((unsigned char)string[0]) & 0xF;
225  cp = (cp << 6) | (string[1] & 0x3F);
226  cp = (cp << 6) | (string[2] & 0x3F);
227  } else if (*len == 4 && IS_CONT(string[1]) && IS_CONT(string[2]) && IS_CONT(string[3])) {
228  cp = string[0] & 0x7;
229  cp = (cp << 6) | (string[1] & 0x3F);
230  cp = (cp << 6) | (string[2] & 0x3F);
231  cp = (cp << 6) | (string[3] & 0x3F);
232  } else {
233  return 0;
234  }
235 
236  /* overlong encodings */
237  if ((cp < 0x80 && *len > 1) ||
238  (cp < 0x800 && *len > 2) ||
239  (cp < 0x10000 && *len > 3)) {
240  return 0;
241  }
242 
243  /* invalid unicode */
244  if (cp > 0x10FFFF) {
245  return 0;
246  }
247 
248  /* surrogate halves */
249  if (cp >= 0xD800 && cp <= 0xDFFF) {
250  return 0;
251  }
252 
253  return 1;
254 }

576 static int parse_utf16(const char **unprocessed, char **processed) {
577  unsigned int cp, lead, trail;
578  int parse_succeeded = 0;
579  char *processed_ptr = *processed;
580  const char *unprocessed_ptr = *unprocessed;
581  unprocessed_ptr++; /* skips u */
582  parse_succeeded = parse_utf16_hex(unprocessed_ptr, &cp);
583  if (!parse_succeeded) {
584  return JSONFailure;
585  }
586  if (cp < 0x80) {
587  processed_ptr[0] = (char)cp; /* 0xxxxxxx */
588  } else if (cp < 0x800) {
589  processed_ptr[0] = ((cp >> 6) & 0x1F) | 0xC0; /* 110xxxxx */
590  processed_ptr[1] = ((cp) & 0x3F) | 0x80; /* 10xxxxxx */
591  processed_ptr += 1;
592  } else if (cp < 0xD800 || cp > 0xDFFF) {
593  processed_ptr[0] = ((cp >> 12) & 0x0F) | 0xE0; /* 1110xxxx */
594  processed_ptr[1] = ((cp >> 6) & 0x3F) | 0x80; /* 10xxxxxx */
595  processed_ptr[2] = ((cp) & 0x3F) | 0x80; /* 10xxxxxx */
596  processed_ptr += 2;
597  } else if (cp >= 0xD800 && cp <= 0xDBFF) { /* lead surrogate (0xD800..0xDBFF) */
598  lead = cp;
599  unprocessed_ptr += 4; /* should always be within the buffer, otherwise previous sscanf would fail */
600  if (*unprocessed_ptr++ != '\\' || *unprocessed_ptr++ != 'u') {
601  return JSONFailure;
602  }
603  parse_succeeded = parse_utf16_hex(unprocessed_ptr, &trail);
604  if (!parse_succeeded || trail < 0xDC00 || trail > 0xDFFF) { /* valid trail surrogate? (0xDC00..0xDFFF) */
605  return JSONFailure;
606  }
607  cp = ((((lead - 0xD800) & 0x3FF) << 10) | ((trail - 0xDC00) & 0x3FF)) + 0x010000;
608  processed_ptr[0] = (((cp >> 18) & 0x07) | 0xF0); /* 11110xxx */
609  processed_ptr[1] = (((cp >> 12) & 0x3F) | 0x80); /* 10xxxxxx */
610  processed_ptr[2] = (((cp >> 6) & 0x3F) | 0x80); /* 10xxxxxx */
611  processed_ptr[3] = (((cp) & 0x3F) | 0x80); /* 10xxxxxx */
612  processed_ptr += 3;
613  } else { /* trail surrogate before lead surrogate */
614  return JSONFailure;
615  }
616  unprocessed_ptr += 3;
617  *processed = processed_ptr;
618  *unprocessed = unprocessed_ptr;
619  return JSONSuccess;
620 }

629  char *output = NULL, *output_ptr = NULL, *resized_output = NULL;
630  output = (char*)parson_malloc(initial_size);
631  if (output == NULL) {
632  goto error;
633  }
634  output_ptr = output;
635  while ((*input_ptr != '\0') && (size_t)(input_ptr - input) < len) {
636  if (*input_ptr == '\\') {
637  input_ptr++;
638  switch (*input_ptr) {
639  case '\"': *output_ptr = '\"'; break;
640  case '\\': *output_ptr = '\\'; break;
641  case '/': *output_ptr = '/'; break;
642  case 'b': *output_ptr = '\b'; break;
643  case 'f': *output_ptr = '\f'; break;
644  case 'n': *output_ptr = '\n'; break;
645  case 'r': *output_ptr = '\r'; break;
646  case 't': *output_ptr = '\t'; break;
647  case 'u':
648  if (parse_utf16(&input_ptr, &output_ptr) == JSONFailure) {
649  goto error;
650  }
651  break;
652  default:
653  goto error;
654  }
655  } else if ((unsigned char)*input_ptr < 0x20) {
656  goto error; /* 0x00-0x19 are invalid characters for json string (http://www.ietf.org/rfc/rfc4627.txt) */
657  } else {
658  *output_ptr = *input_ptr;
659  }
660  output_ptr++;
661  input_ptr++;
662  }
663  *output_ptr = '\0';
664  /* resize to new length */
665  final_size = (size_t)(output_ptr-output) + 1;
666  /* todo: don't resize if final_size == initial_size */
667  resized_output = (char*)parson_malloc(final_size);
668  if (resized_output == NULL) {
669  goto error;
670  }
671  memcpy(resized_output, output, final_size);
672  parson_free(output);
673  return resized_output;
674 error:
675  parson_free(output);
676  return NULL;
677 }

717 static JSON_Value * parse_object_value(const char **string, size_t nesting) {
718  JSON_Value *output_value = NULL, *new_value = NULL;
719  JSON_Object *output_object = NULL;
720  char *new_key = NULL;
721  output_value = json_value_init_object();
722  if (output_value == NULL) {
723  return NULL;
724  }
725  if (**string != '{') {
726  json_value_free(output_value);
727  return NULL;
728  }
729  output_object = json_value_get_object(output_value);
730  SKIP_CHAR(string);
731  SKIP_WHITESPACES(string);
732  if (**string == '}') { /* empty object */
733  SKIP_CHAR(string);
734  return output_value;

736  while (**string != '\0') {
737  new_key = get_quoted_string(string);
738  if (new_key == NULL) {

745  json_value_free(output_value);
746  return NULL;

748  SKIP_CHAR(string);

750  if (new_value == NULL) {
751  parson_free(new_key);
752  json_value_free(output_value);

756  parson_free(new_key);

758  json_value_free(output_value);

763  if (**string != ',') {
764  break;
765  }
766  SKIP_CHAR(string);
767  SKIP_WHITESPACES(string);
768  }
769  SKIP_WHITESPACES(string);
770  if (**string != '}' || /* Trim object after parsing is over */
771  json_object_resize(output_object, json_object_get_count(output_object)) == JSONFailure) {
772  json_value_free(output_value);
773  return NULL;
774  }
775  SKIP_CHAR(string);
776  return output_value;
777 }

1914  JSON_Value_Type schema_type = JSONError, value_type = JSONError;

1917  if (schema == NULL || value == NULL) {
1918  return JSONFailure;
1919  }
1920  schema_type = json_value_get_type(schema);
1921  value_type = json_value_get_type(value);
1922  if (schema_type != value_type && schema_type != JSONNull) { /* null represents all values */

1924  }

1928  value_array = json_value_get_array(value);

1931  return JSONSuccess; /* Empty array allows all types */

1949  return JSONFailure; /* Tested object mustn't have less name-value pairs than schema */

1952  key = json_object_get_name(schema_object, i);
